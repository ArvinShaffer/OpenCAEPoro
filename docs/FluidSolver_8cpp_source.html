181bool OCP_IMPEC::UpdateProperty(Reservoir& rs, OCP_DBL& dt)
182{
183 // first check : Pressure check
184 OCP_INT flagCheck = rs.CheckP();
185 switch (flagCheck) {
186 case 1:
187 cout << "well change" << endl;
188 dt /= 2;
189 return false;
190 case 2:
191 cout << "well change" << endl;
192 dt /= 2;
193 return false;
194 default:
195 break;
196 }
197
198 rs.CalFLux();
199
200 // second check : CFL check
201 OCP_DBL cfl = rs.CalCFL01(dt);
202 if (cfl > 1) {
203 dt /= 2;
204 rs.ResetVal();
205 cout << "CFL is too big" << endl;
206 return false;
207 }
208
209 rs.MassConseve(dt);
210
211 // third check: Ni check
212 if (!rs.CheckNi()) {
213 dt /= 2;
214 rs.ResetVal01();
215 cout << "Negative Ni occurs\n";
216 return false;
217 }
218
219 rs.CalFlashIMPEC();
220 rs.CalVpore();
221
222 // fouth check: Volume error check
223 if (!rs.CheckVe(0.01)) {
224 dt /= 2;
225 rs.ResetVal02();
226 return false;
227 }
228
229 rs.CalKrPc();
230 rs.CalConnFlux();
231 return true;
232}
