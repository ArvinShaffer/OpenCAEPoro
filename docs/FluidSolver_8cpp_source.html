125bool OCP_IMPEC::UpdateProperty(Reservoir& rs, OCP_DBL& dt)
126{
127 // first check : Pressure check
128 OCP_INT flagCheck = rs.CheckP();
129 switch (flagCheck) {
130 case 1:
131 cout << "well change" << endl;
132 dt /= 2;
133 return false;
134 case 2:
135 cout << "well change" << endl;
136 dt /= 2;
137 return false;
138 default:
139 break;
140 }
141
142 rs.CalFLux();
143
144 // second check : CFL check
145 OCP_DBL cfl = rs.CalCFL01(dt);
146 if (cfl > 1) {
147 dt /= 2;
148 rs.ResetVal();
149 cout << "CFL is too big" << endl;
150 return false;
151 }
152
153 rs.MassConseve(dt);
154
155 // third check: Ni check
156 if (!rs.CheckNi()) {
157 dt /= 2;
158 rs.ResetVal01();
159 cout << "Negative Ni occurs\n";
160 return false;
161 }
162
163 rs.CalFlashIMPEC();
164 rs.CalVpore();
165
166 // fouth check: Volume error check
167 if (!rs.CheckVe(0.01)) {
168 cout << "###WARNING: volume error is too big\n";
169 dt /= 2;
170 rs.ResetVal02();
171 return false;
172 }
173
174 rs.CalKrPc();
175 rs.CalConnFlux();
176 return true;
177}
