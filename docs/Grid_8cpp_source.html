20void Grid::CalDepthV()
21{
22 depth.resize(numGrid, 0);
23 OCP_USI nxny = nx * ny;
24 // 0th layer
25 for (USI j = 0; j < ny; j++) {
26 for (USI i = 0; i < nx; i++) {
27 OCP_USI id = j * nx + i;
28 depth[id] = tops[id] + dz[id] / 2;
29 }
30 }
31 // 1th - (nz-1)th layer
32 for (USI k = 1; k < nz; k++) {
33 OCP_USI knxny = k * nxny;
34 for (USI j = 0; j < ny; j++) {
35 for (USI i = 0; i < nx; i++) {
36 OCP_USI id = knxny + j * nx + i;
37 depth[id] = depth[id - nxny] + dz[id - nxny] / 2 + dz[id] / 2;
38 }
39 }
40 }

47void Grid::CalActiveGrid(const OCP_DBL& e1, const OCP_DBL& e2)
48{
49 activeMap_B2G.reserve(numGrid);
50 activeMap_G2B.resize(numGrid);
51 OCP_USI count = 0;
52 for (OCP_USI n = 0; n < numGrid; n++) {
53 if (poro[n] * ntg[n] < e1 || dx[n] * dy[n] * dz[n] < e2) {
54 activeMap_G2B[n] = GB_Pair(false, 0);
55 continue;
56 }
57 activeMap_B2G.push_back(n);
58 activeMap_G2B[n] = GB_Pair(true, count);
59 count++;
60 }
61 activeGridNum = count;
62 cout << (numGrid - activeGridNum) / numGrid << "% grids is inactive" << endl;
63 cout << "Grid::calActiveBulk" << endl;
64}
