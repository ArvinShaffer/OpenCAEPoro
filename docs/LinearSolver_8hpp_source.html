146template <typename T> void Solver<T>::AssembleMat_Fasp()
147{
148 // b & x
149 b_Fasp.row = dim;
150 b_Fasp.val = b.data();
151 x_Fasp.row = dim;
152 x_Fasp.val = u.data();
153 // A
154 OCP_USI nnz = 0;
155 for (OCP_USI i = 0; i < dim; i++) {
156 nnz += colId[i].size();
157 }
158 A_Fasp = fasp_dcsr_create(dim, dim, nnz);
159 // IA
160
161 OCP_USI count = 0;
162 for (OCP_USI i = 1; i < dim + 1; i++) {
163 USI nnz_Row = colId[i - 1].size();
164 A_Fasp.IA[i] = A_Fasp.IA[i - 1] + nnz_Row;
165
166 for (USI j = 0; j < nnz_Row; j++) {
167 A_Fasp.JA[count] = colId[i - 1][j];
168 A_Fasp.val[count] = val[i - 1][j];
169 count++;
170 }
171 }
172}

174template <typename T> int Solver<T>::FaspSolve()
175{
176 int status = FASP_SUCCESS;
177
178 const int print_level = inParam.print_level;
179 const int solver_type = inParam.solver_type;
180 const int precond_type = inParam.precond_type;
181 const int output_type = inParam.output_type;
182
183 if (output_type) {
184 const char* outputfile = "out/Solver.out";
185 printf("Redirecting outputs to file: %s ...\n", outputfile);
186 freopen(outputfile, "w", stdout); // open a file for stdout
187 }
188
189 // Preconditioned Krylov methods
190 if (solver_type >= 1 && solver_type <= 20) {
191
192 // Using no preconditioner for Krylov iterative methods
193 if (precond_type == PREC_NULL) {
194 status = fasp_solver_dcsr_krylov(&A_Fasp, &b_Fasp, &x_Fasp, &itParam);
195 }
196
197 // Using diag(A) as preconditioner for Krylov iterative methods
198 else if (precond_type == PREC_DIAG) {
199 status = fasp_solver_dcsr_krylov_diag(&A_Fasp, &b_Fasp, &x_Fasp, &itParam);
200 }
201
202 // Using AMG as preconditioner for Krylov iterative methods
203 else if (precond_type == PREC_AMG || precond_type == PREC_FMG) {
204 if (print_level > PRINT_NONE) fasp_param_amg_print(&amgParam);
205 status = fasp_solver_dcsr_krylov_amg(&A_Fasp, &b_Fasp, &x_Fasp, &itParam,
206 &amgParam);
207 }
208
209 // Using ILU as preconditioner for Krylov iterative methods
210 else if (precond_type == PREC_ILU) {
211 if (print_level > PRINT_NONE) fasp_param_ilu_print(&iluParam);
212 status = fasp_solver_dcsr_krylov_ilu(&A_Fasp, &b_Fasp, &x_Fasp, &itParam,
213 &iluParam);
214 }
215
216 // Undefined iterative methods
217 else {
218 printf("### ERROR: Wrong preconditioner type %d!!!\n", precond_type);
219 status = ERROR_SOLVER_PRECTYPE;
220 }
221
222 }
223
224 // AMG as the iterative solver
225 else if (solver_type == SOLVER_AMG) {
226 if (print_level > PRINT_NONE) fasp_param_amg_print(&amgParam);
227 fasp_solver_amg(&A_Fasp, &b_Fasp, &x_Fasp, &amgParam);
228 }
229
230 // Full AMG as the iterative solver
231 else if (solver_type == SOLVER_FMG) {
232 if (print_level > PRINT_NONE) fasp_param_amg_print(&amgParam);
233 fasp_solver_famg(&A_Fasp, &b_Fasp, &x_Fasp, &amgParam);
234 }
235
236#if WITH_MUMPS // use MUMPS directly
237 else if (solver_type == SOLVER_MUMPS) {
238 status = fasp_solver_mumps(&A_Fasp, &b_Fasp, &x_Fasp, print_level);
239 if (status >= 0) status = 1; // Direct solver returns 1
240 }
241#endif
242
243#if WITH_SuperLU // use SuperLU directly
244 else if (solver_type == SOLVER_SUPERLU) {
245 status = fasp_solver_superlu(&A_Fasp, &b_Fasp, &x_Fasp, print_level);
246 if (status >= 0) status = 1; // Direct solver returns 1
247 }
248#endif
249
250#if WITH_UMFPACK // use UMFPACK directly
251 else if (solver_type == SOLVER_UMFPACK) {
252 // Need to sort the matrix A for UMFPACK to work
253 dCSRmat A_trans = fasp_dcsr_create(A_Fasp.row, A_Fasp.col, A_Fasp.nnz);
254 fasp_dcsr_transz(&A_Fasp, NULL, &A_trans);
255 fasp_dcsr_sort(&A_trans);
256 status = fasp_solver_umfpack(&A_trans, &b_Fasp, &x_Fasp, print_level);
257 fasp_dcsr_free(&A_trans);
258 if (status >= 0) status = 1; // Direct solver returns 1
259 }
260#endif
261
262#ifdef WITH_PARDISO // use PARDISO directly
263 else if (solver_type == SOLVER_PARDISO) {
264 fasp_dcsr_sort(&A_Fasp);
265 status = fasp_solver_pardiso(&A_Fasp, &b_Fasp, &x_Fasp, print_level);
266 if (status >= 0) status = 1; // Direct solver returns 1
267 }
268#endif
269
270 else {
271 printf("### ERROR: Wrong solver type %d!!!\n", solver_type);
272 status = ERROR_SOLVER_TYPE;
273 }
274
275 if (status < 0) {
276 printf("### ERROR: Solver failed! Exit status = %d.\n\n", status);
277 }
278
279 if (output_type) fclose(stdout);
280 return status;
281}

286 string FileA = solveDir + fileA;
287 string Fileb = solveDir + fileb;

293 for (OCP_USI i = 0; i < A_Fasp.row + 1; i++) outA << A_Fasp.IA[i] + 1 << endl;
294 for (OCP_USI i = 0; i < nnz0; i++) outA << A_Fasp.JA[i] + 1 << endl;
295 for (OCP_USI i = 0; i < nnz0; i++) outA << A_Fasp.val[i] << endl;
296 outA.close();
297 // out rhs
298 ofstream outb(Fileb);
299 if (!outb.is_open()) cout << "Can not open " << Fileb << endl;
300 outb << b_Fasp.row << endl;
301 for (OCP_USI i = 0; i < b_Fasp.row; i++) outb << b_Fasp.val[i] << endl;
302 outb.close();
303}

337template <typename T> void Solver<T>::ReadParam_Fasp()
338{
339 string file = solveDir + solveFile;
340 InitParam_Fasp(); // Set default solver parameters
341 std::ifstream ifs(file);
342 if (!ifs.is_open()) {
343 std::cout << "The input file " << file << " is missing!" << std::endl;
344 file = solveDir + "../conf/csr.dat";
345 ifs.open(file);
346 if (!ifs.is_open()) {
347 std::cout << "The input file " << file << " is missing!" << std::endl;
348 std::cout << "Using the default parameters of FASP" << std::endl;
349 } else {
350 ifs.close();
351 std::cout << "Using the input file " << file << std::endl;
352 fasp_param_input(file.data(), &inParam);
353 }
354 } else {
355 ifs.close(); // if file has been opened, close it first
356 fasp_param_input(file.data(), &inParam);
357 }
358 fasp_param_init(&inParam, &itParam, &amgParam, &iluParam, &swzParam);
359}
