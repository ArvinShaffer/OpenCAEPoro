4void BOMixture::BOFlash_Sj_OW(const OCP_DBL& Pin, const OCP_DBL* Sjin,
5 const OCP_DBL& Vpore)
6{
7 P = Pin;
8 S[1] = Sjin[1];
9 // Water Properties
10 PVTW.Eval_All(0, P, data, cdata);
11 OCP_DBL Pw0 = data[0];
12 OCP_DBL bw0 = data[1];
13 OCP_DBL cbw = data[2];
14 OCP_DBL bw = bw0 * (1 - cbw * (P - Pw0));
15 OCP_DBL bwp = -cbw * bw0;
16
17 mu[1] = data[3];
18 xi[1] = 1 / (bw * CONV1);
19 rho[1] = std_RhoW / bw;
20
21 Ni[1] = Vpore * S[1] * xi[1];
22
23 // Oil Properties
24 PVDO.Eval_All(0, P, data, cdata);
25 OCP_DBL bo = data[1];
26 OCP_DBL bop = cdata[1];
27
28 mu[0] = data[2];
29 xi[0] = 1 / (CONV1 * bo);
30 rho[0] = std_RhoO / bo;
31 Ni[0] = Vpore * (1 - S[1]) * xi[0];
32
33 cij[0 * 2 + 0] = 1; cij[0 * 2 + 1] = 0;
34 cij[1 * 2 + 0] = 0; cij[1 * 2 + 1] = 1;
35
36 v[0] = CONV1 * Ni[0] * bo;
37 v[1] = CONV1 * Ni[1] * bw;
38 vf = v[0] + v[1];
39 S[0] = v[0] / vf;
40 S[1] = v[1] / vf;
41 vfi[0] = CONV1 * bo;
42 vfi[1] = CONV1 * bw;
43 vfp = CONV1 * Ni[0] * bop + CONV1 * Ni[1] * bwp;

53 Ni[0] = Niin[0];
54 Ni[1] = Niin[1];
55 Nt = Ni[0] + Ni[1];
56
57 // Water Properties
58 PVTW.Eval_All(0, P, data, cdata);

60 OCP_DBL bw0 = data[1];
61 OCP_DBL cbw = data[2];

67 rho[1] = std_RhoW / bw;

72
73 // Oil Properties

81
82 muP[0] = cdata[2];
83 xiP[0] = -xi[0] * bop / bo;
84 rhoP[0] = -rho[0] * bop / bo;
85
86 cij[0 * 2 + 0] = 1; cij[0 * 2 + 1] = 0;
87 cij[1 * 2 + 0] = 0; cij[1 * 2 + 1] = 1;
88

90 v[1] = CONV1 * Ni[1] * bw;

92 S[0] = v[0] / vf;
93 S[1] = v[1] / vf;

97

99 dSec_dPri[1] = (CONV1 * bo - S[0] * vfi[0]) / vf; // dSo / dNo
100 dSec_dPri[2] = -S[0] * vfi[1] / vf; // dSo / dNw
101
102 dSec_dPri[3] = (CONV1 * Ni[1] * bwp - S[1] * vfp) / vf; // dSw / dP
103 dSec_dPri[4] = -S[1] * vfi[0] / vf; // dSw / dNo
104 dSec_dPri[5] = (CONV1 * bw - S[1] * vfi[1]) / vf; // dSw / dNw
