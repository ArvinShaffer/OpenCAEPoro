53bool OCP_IMPEC::UpdateProperty(Reservoir& rs, OCP_DBL& dt)
54{
55 // first check : Pressure check
56 OCP_INT flagCheck = rs.CheckP();
57 switch (flagCheck) {
58 case 1:
59 cout << "well change" << endl;
60 dt /= 2;
61 return false;
62 case 2:
63 cout << "well change" << endl;
64 dt /= 2;
65 return false;
66 default:
67 break;
68 }
69
70 rs.CalFLuxIMPEC();
71
72 // second check : CFL check
73 OCP_DBL cfl = rs.CalCFL01IMPEC(dt);
74 if (cfl > 1) {
75 dt /= 2;
76 rs.ResetVal01IMPEC();
77 cout << "CFL is too big" << endl;
78 return false;
79 }
80
81 rs.MassConseveIMPEC(dt);
82
83 // third check: Ni check
84 if (!rs.CheckNi()) {
85 dt /= 2;
86 rs.ResetVal02IMPEC();
87 cout << "Negative Ni occurs\n";
88 return false;
89 }
90
91 rs.CalFlashIMPEC();
92 rs.CalVpore();
93
94 // fouth check: Volume error check
95 if (!rs.CheckVe(0.01)) {
96 dt /= 2;
97 rs.ResetVal03IMPEC();
98 return false;
99 }
100
101 rs.CalKrPc();
102 rs.CalConnFluxIMPEC();
103 return true;
104}
