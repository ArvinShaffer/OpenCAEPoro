45void OCP_IMPES::GoOneStep(Reservoir& rs, OCP_Control& ctrl)
46{
47 OCP_DBL ve = 0.01;
48 OCP_DBL cfl = 1;
49 int flagCheck = 0;
50 double& dt = ctrl.current_dt;
51
52 rs.wellgroup.PrepareWell(rs.bulk);
53
54 cfl = rs.CalCFL(dt);
55 if (cfl > 1) dt /= (cfl + 1);
56
57 while (true) {
58 SolveP(rs, ctrl, dt);
59
60 // first check : Pressure check
61 flagCheck = rs.CheckP();
62 if (flagCheck == 1) {
63 dt /= 2;
64 continue;
65 } else if (flagCheck == 2) {
66 continue;
67 }
68
69 rs.conn.CalFlux(rs.bulk);
70 rs.wellgroup.CalFlux(rs.bulk);
71
72 // second check : cfl check
73 cfl = rs.CalCFL(dt);
74 if (cfl > 1) {
75 dt /= 2;
76 rs.ResetVal01();
77 continue;
78 }
79
80 rs.conn.MassConserve(rs.bulk, dt);
81 rs.wellgroup.MassConserve(rs.bulk, dt);
82
83 // third check: Ni check
84 if (!rs.CheckNi()) {
85 dt /= 2;
86 rs.ResetVal01();
87 continue;
88 }
89
90 rs.bulk.FlashNi();
91 rs.bulk.CalVporo();
92
93 // fouth check: Volume error check
94 if (!rs.CheckVe(ve)) {
95 dt /= 2;
96 rs.ResetVal02();
97 continue;
98 }
99
100 rs.bulk.CalKrPc();
101 rs.conn.CalFlux(rs.bulk);
102
103 break;
104 }
105
106 rs.wellgroup.CalIPRT(rs.bulk, dt);
107 ctrl.tstep += 1;
108 ctrl.iterNR = 1;
109 ctrl.iterNR_total += 1;
110
111 rs.bulk.CalMaxChange();
112 ctrl.SetNextTstep(rs);
113 rs.bulk.SetLastStep();
114 rs.wellgroup.SetLastStep();
115}
