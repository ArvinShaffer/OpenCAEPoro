93 if (val >= data[j][bId]) {
94 for (USI i = bId + 1; i < nRow; i++) {
95 if (val < data[j][i]) {
96 bId = i - 1;
97 for (USI k = 0; k < nCol; k++) {
98 slope[k] = (data[k][bId + 1] - data[k][bId]) /
99 (data[j][bId + 1] - data[j][bId]);
100 outdata[k] = data[k][bId] + slope[k] * (val - data[j][bId]);
101 }
102 return bId;
103 }
104 }
105 for (USI k = 0; k < nCol; k++) {
106 slope[k] = 0;
107 outdata[k] = data[k].back();

112 bId = i;
