23void Solver::RunSimulation(Reservoir& rs, OCP_Control& ctrl, OCP_Output& output)
24{
25 GetWallTime timer;
26 timer.Start();
27
28 USI numTSteps = ctrl.GetNumTSteps();
29 output.PrintInfoSched(rs, ctrl, timer.Stop());
30
31 for (USI d = 0; d < numTSteps - 1; d++) {
32 rs.ApplyControl(d);
33 ctrl.ApplyControl(d);
34 ctrl.InitTime(d);
35 while (!ctrl.IsCriticalTime(d + 1)) {
36 GoOneStep(rs, ctrl);
37 output.SetVal(rs, ctrl);
38 }
39 output.PrintInfoSched(rs, ctrl, timer.Stop());
40 }
41
42 ctrl.RecordTotalTime(timer.Stop() / 1000);
43}

46void Solver::GoOneStep(Reservoir& rs, OCP_Control& ctrl)
47{
48 OCP_DBL& dt = ctrl.GetCurDt();
49
50 // Prepare for time marching
51 Prepare(rs, dt);
52
53 // Time marching with adaptive time stepsize
54 while (true) {
55 if (dt < MIN_TIME_CURSTEP) OCP_ABORT("Time stepsize is too small!");
56 AssembleSolve(rs, ctrl, dt);
57 if (!UpdateProperty(rs, dt)) continue;
58 if (FinishNR()) break;
59 }
60
61 // Finish current time step
62 FinishStep(rs, ctrl);
63}
