158void WellGroup::CalIPRT(const Bulk& myBulk, OCP_DBL dt)
159{
160 FGIR = 0;
161 FWIR = 0;
162 FOPR = 0;
163 FGPR = 0;
164 FWPR = 0;
165 for (USI w = 0; w < numWell; w++) {
166 wellGroup[w].WGIR = 0;
167 wellGroup[w].WWIR = 0;
168 wellGroup[w].WOPR = 0;
169 wellGroup[w].WGPR = 0;
170 wellGroup[w].WWPR = 0;
171
172 if (wellGroup[w].WellState()) {
173 if (wellGroup[w].WellType() == PROD) {
174 wellGroup[w].CalProdQi_Blk(myBulk, dt);
175 } else {
176 wellGroup[w].CalInjQi_Blk(myBulk, dt);
177 }
178 }
179 FGIR += wellGroup[w].WGIR;
180 FWIR += wellGroup[w].WWIR;
181 FOPR += wellGroup[w].WOPR;
182 FGPR += wellGroup[w].WGPR;
183 FWPR += wellGroup[w].WWPR;
184 }
185 FGIT += FGIR * dt;
186 FWIT += FWIR * dt;
187 FOPT += FOPR * dt;
188 FGPt += FGPR * dt;
189 FWPT += FWPR * dt;
190}

192OCP_INT WellGroup::CheckP(const Bulk& myBulk)
193{
194 // 0 : All correct
195 // 1 : negative P, cut the timestep and resolve
196 // 2.1 : change well mode to BHP, resolve
197 // 2.2 : crossflow happens, then close corresponding perf, resolve
198 bool flag2 = false;
199 bool flag3 = false;
200
201 for (USI w = 0; w < numWell; w++) {
202 if (wellGroup[w].WellState()) {
203
204 OCP_INT flag = wellGroup[w].CheckP(myBulk);
205#ifdef _DEBUG
206 // wellGroup[w].ShowPerfStatus();
207#endif // _DEBUG
208 switch (flag) {
209 case 1:
210 return 1;
211 case 2:
212 flag2 = true;
213 break;
214 case 3:
215 flag3 = true;
216 break;
217 default:
218 break;
219 }
220 }
221 }
222
223 if (flag2 || flag3) return 2;
224
225 return 0;
226}
